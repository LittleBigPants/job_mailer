      Mock GET https://example.com → 200, body='&lt;p&gt;Reach us at hello@example.com&lt;/p&gt;'
      Assert: record.email_found == "hello@example.com"

    test_contact_fallback:
      Mock GET https://example.com → 200, body='&lt;p&gt;No email here&lt;/p&gt;'
      Mock GET https://example.com/contact → 200, body='&lt;a href="mailto:contact@example.com"&gt;Contact&lt;/a&gt;'
      Assert: record.email_found == "contact@example.com"

    test_about_fallback:
      Mock GET https://example.com → 200, body='&lt;p&gt;No email here&lt;/p&gt;'
      Mock GET https://example.com/contact → 200, body='&lt;p&gt;No email here&lt;/p&gt;'
      Mock GET https://example.com/about → 200, body='&lt;a href="mailto:team@example.com"&gt;Team&lt;/a&gt;'
      Assert: record.email_found == "team@example.com"

    test_no_email_found:
      Mock all three URLs → 200, body='&lt;p&gt;No email here&lt;/p&gt;'
      Assert: record.email_found == ""
      Assert: record.status == Status.NO_EMAIL_FOUND

    test_scrape_failed:
      All three GET calls raise httpx.ConnectError (use httpx_mock with exc=httpx.ConnectError("timeout"))
      Assert: record.status == Status.SCRAPE_FAILED

    test_email_priority:
      Mock GET https://example.com → 200, body contains both 'info@example.com' (text) and 'jobs@example.com' (mailto: link)

    test_never_return_excluded:
      Mock GET https://example.com → 200, body='&lt;a href="mailto:noreply@example.com"&gt;x&lt;/a&gt;'

      Assert: record.status == Status.NO_EMAIL_FOUND
