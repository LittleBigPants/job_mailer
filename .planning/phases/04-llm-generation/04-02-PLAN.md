CLICHE_DENY_LIST = [

    "quick question",
    "synergy",
    "touch base",

]
def _word_count(text: str) -> int: return len(text.split())
def _has_brackets(text: str) -> bool: return bool(re.search(r'\[.+?\]', text))
def _has_cliche(text: str) -> bool:
    lower = text.lower()
